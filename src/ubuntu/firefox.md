```shell
sudo add-apt-repository ppa:ubuntu-mozilla-daily/ppa
sudo apt update
sudo apt install firefox-trunk install firefox-trunk-locale-pl
```

```
sudo apt install --only-upgrade firefox-trunk   firefox-trunk-locale-pl
```
